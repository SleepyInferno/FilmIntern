```typescript
describe('DATABASE_PATH configuration', () => {

    vi.resetModules();
```

```typescript
const nextConfig: NextConfig = {

```

This adds `output: 'standalone'` (DOCK-02) and `outputFileTracingIncludes` to ensure the better-sqlite3 native `.node` binary is included in the standalone trace output.

**Step 2 — Create src/lib/version.ts:**

```typescript
export const APP_VERSION = '0.1.0';
```

This is the simplest approach for a personal tool. `process.env.npm_package_version` is NOT available in standalone mode, and importing package.json has fragile relative paths. A constant in a shared file is easy to update when version changes.

**Step 3 — Create src/app/api/health/route.ts:**

```typescript
import { NextResponse } from 'next/server';
import { APP_VERSION } from '@/lib/version';

export const dynamic = 'force-dynamic';

export async function GET() {
  try {
    const { db } = await import('@/lib/db');
    // Lightweight connectivity check — if this throws, DB is unreachable
    db.listProjects();

    return NextResponse.json({
      status: 'ok',
      version: APP_VERSION,
      db: 'connected',
    });
  } catch {
    return NextResponse.json(
      { status: 'error', db: 'failed' },
      { status: 503 },
    );
  }
}
```

Notes:
- `export const dynamic = 'force-dynamic'` prevents Next.js from caching the health check response at build time
- Uses dynamic import of `@/lib/db` to avoid module-level initialization at import time — the health check should only touch the DB when called
- `db.listProjects()` is used as the connectivity check because it runs a real SELECT query; if the DB file is missing or corrupt, it will throw
- Returns 200 with `{ status: "ok", version: "0.1.0", db: "connected" }` on success
- Returns 503 with `{ status: "error", db: "failed" }` on failure
